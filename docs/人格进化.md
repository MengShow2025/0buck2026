async def update_dumbo_soul(user_id, feedback_type, comment=None):
    """
    根据用户反馈自动进化阿呆
    feedback_type: 'like', 'dislike', 'correction'
    """
    current_soul = await db.get_user_soul(user_id)
    
    if feedback_type == 'like':
        current_soul['affinity'] += 5
        # 好感度达标，自动切换称呼
        if current_soul['affinity'] > 90:
            current_soul['call_user_as'] = "知己"
            current_soul['style'] += "可以更随意、更亲昵地开玩笑。"
            
    elif feedback_type == 'dislike' and "太啰嗦" in comment:
        # 自动调整语言密度
        current_soul['chat_density'] = "极简短句"
        current_soul['style'] = "直接给结论，减少语气助词。"
        
    await db.save_user_soul(user_id, current_soul)
```
